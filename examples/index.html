Start
快速开始
import * as cigvis from 'cigvis';

const { data, shape } = await cigvis.io.readVolume(
  'seismic.bin', [192, 200, 240]
);

const nodes = cigvis.createSlices({
  data, shape,
  pos: { z: 120 },
  cmap: 'petrel',
});

cigvis.plot3D({ nodes, container: '#viewer' });
